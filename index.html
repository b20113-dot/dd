    let playerMarbles = 12;
    let compMarbles = 12;
    let currentTurn = 'attack';
    let maxTime = 7;
    let timeLeft = 7;
    let timerInterval;
    let isProcessing = false;
    // 유저의 행동 패턴 기억 (최근 3번의 방어 유형 기록 - 'odd' 또는 'even')
    let playerPattern = [];
    function startTimer() {
    clearInterval(timerInterval);
    // 난이도 메커니즘 1: 패닉 타임어택 (구슬 5개 이하 시 3초로 대폭 제한)
    if (playerMarbles <= 5) {
    maxTime = 3;
    document.getElementById('timerWarn').style.display = 'inline';
    } else {
    maxTime = 7;
    document.getElementById('timerWarn').style.display = 'none';
    }
    timeLeft = maxTime;
    document.getElementById('timeDisplay').innerText = timeLeft;
    let timerBar = document.getElementById('timerBar');
    timerBar.style.transition = 'none';
    timerBar.style.width = '100%';
    timerBar.style.backgroundColor = '#8b0000';
    setTimeout(() => {
    timerBar.style.transition = `width 1s linear, background-color 0.5s`;
    }, 50);
    timerInterval = setInterval(() => {
    if(isProcessing) return;
    timeLeft--;
    document.getElementById('timeDisplay').innerText = timeLeft;
    let percent = (timeLeft / maxTime) * 100;
    timerBar.style.width = percent + '%';
    if(timeLeft <= 2) {
    timerBar.style.backgroundColor = '#ff0000';
    }
    if (timeLeft <= 0) {
    clearInterval(timerInterval);
    handleTimeout();
    }
    }, 1000);
    }
    function handleTimeout() {
    isProcessing = true;
    let msg = "";
    if (currentTurn === 'attack') {
    msg = `<span style="color:#ff3333; font-size:16px;">[ 경고 ] 의사결정 지체.</span><br><span style="color:#444">공격 자격 박탈. 단두대가 한 칸 내려옵니다.</span>`;
    document.getElementById('message').innerHTML = msg;
    currentTurn = 'defend';
    } else {
    let forcedHide = 1;
    document.getElementById('hideAmount').value = forcedHide;
    msg = `<span style="color:#ff3333; font-size:16px;">[ 경고 ] 무단 기동 정지.</span><br><span style="color:#444">벌벌 떨다 구슬 1개만 쥐었습니다. 강제 검문을 시작합니다.</span>`;
    document.getElementById('message').innerHTML = msg;
    setTimeout(() => {
    isProcessing = false;
    playDefend(true);
    }, 1500);
    return;
    }
    setTimeout(() => {
    updateUI();
    isProcessing = false;
    }, 2500);
    }
    function updateUI() {
    document.getElementById('playerMarbles').innerText = playerMarbles;
    document.getElementById('compMarbles').innerText = compMarbles;
    if (playerMarbles <= 5) {
    document.body.classList.add('danger-mode');
    document.getElementById('playerLabel').classList.add('player-danger');
    } else {
    document.body.classList.remove('danger-mode');
    document.getElementById('playerLabel').classList.remove('player-danger');
    }
    let inputField = document.getElementById('hideAmount');
    inputField.max = playerMarbles;
    if(parseInt(inputField.value) > playerMarbles) inputField.value = playerMarbles;
    if (currentTurn === 'attack') {
    document.getElementById('attackZone').style.display = 'block';
    document.getElementById('defendZone').style.display = 'none';
    } else {
    document.getElementById('attackZone').style.display = 'none';
    document.getElementById('defendZone').style.display = 'block';
    }
    startTimer();
    }
    function playAttack(playerGuess) {
    if(isProcessing) return;
    isProcessing = true;
    clearInterval(timerInterval);
    // 난이도 메커니즘 2: 매치 후반부로 갈수록 간수가 쥐는 판돈 상한선 상승
    let upperLimit = Math.min(compMarbles, 8);
    let compHidden = Math.floor(Math.random() * upperLimit) + 1;
    let isCompOdd = (compHidden % 2 !== 0);
    let isPlayerGuessOdd = (playerGuess === 'odd');
    let msg = `간수가 펼친 손바닥에는 <strong>${compHidden}개(${isCompOdd ? '홀' : '짝'})</strong>가 놓여있었습니다.<br>`;
    if (isCompOdd === isPlayerGuessOdd) {
    let taken = compHidden;
    if (compMarbles - taken < 0) taken = compMarbles;
    playerMarbles += taken;
    compMarbles -= taken;
    msg += `<span style="color:#ff3333; font-size:16px;">[ 약탈 성공 ] 간수의 주머니에서 ${taken}개의 생명을 찢어왔습니다.</span>`;
    } else {
    msg += `<span style="color:#555">예상이 빗나갔습니다. 허공에 칼질을 한 채 차례가 넘어갑니다.</span>`;
    }
    document.getElementById('message').innerHTML = msg;
    if (checkGameOver()) return;
    currentTurn = 'defend';
    setTimeout(() => {
    updateUI();
    isProcessing = false;
    }, 2500);
    }
    function playDefend(isTimeout = false) {
    if(isProcessing && !isTimeout) return;
    isProcessing = true;
    clearInterval(timerInterval);
    let playerHidden = parseInt(document.getElementById('hideAmount').value);
    if (isNaN(playerHidden) || playerHidden < 1 || playerHidden > playerMarbles) {
    alert(`가진 수명을 넘어서서 사기 칠 생각 마십시오. (1~${playerMarbles}개)`);
    isProcessing = false;
    startTimer();
    return;
    }
    let isPlayerOdd = (playerHidden % 2 !== 0);
    let playerCurrentChoice = isPlayerOdd ? 'odd' : 'even';
    // 난이도 메커니즘 3: 처형자의 예측 저격 알고리즘 수행
    let compGuess = 'odd';
    if (playerPattern.length >= 2) {
    // 유저가 직전에 무엇을 많이 냈는지 카운트하여 저격
    let oddCount = playerPattern.filter(x => x === 'odd').length;
    let evenCount = playerPattern.filter(x => x === 'even').length;
    if (oddCount > evenCount) compGuess = 'odd';
    else if (evenCount > oddCount) compGuess = 'even';
    else compGuess = Math.random() < 0.5 ? 'odd' : 'even';
    } else {
    compGuess = Math.random() < 0.5 ? 'odd' : 'even';
    }
    // 패턴 배열 업데이트 (최근 3개만 유지)
    playerPattern.push(playerCurrentChoice);
    if(playerPattern.length > 3) playerPattern.shift();
    let isCompGuessOdd = (compGuess === 'odd');
    let msg = document.getElementById('message').innerHTML;
    if(!isTimeout) msg = "";
    else msg += "<br><br>";
    msg += `처형자가 당신의 영혼을 꿰뚫어 보듯 <strong>"${compGuess === 'odd' ? '홀' : '짝'}"</strong>을 선고합니다.<br>`;
    if (isPlayerOdd === isCompGuessOdd) {
    // 난이도 메커니즘 4: 걸리면 2배 압수 페널티
    let lost = playerHidden * 2;
    if (playerMarbles - lost < 0) lost = playerMarbles;
    playerMarbles -= lost;
    compMarbles += lost;
    msg += `<span style="color:#ff3333; font-size:16px;">[ 간파당함 - 2배 징벌 ] 처형자가 당신의 버릇을 읽었습니다! 수명 ${lost}조각을 뜯깁니다.</span>`;
    } else {
    msg += `<span style="color:#666">그가 심리전에서 헛다리를 짚었습니다. 간신히 피 냄새를 흘리며 살아남았습니다.</span>`;
    }
    document.getElementById('message').innerHTML = msg;
    if (checkGameOver()) return;
    currentTurn = 'attack';
    setTimeout(() => {
    updateUI();
    isProcessing = false;
    }, 2500);
    }
    function checkGameOver() {
    if (playerMarbles <= 0) {
    clearInterval(timerInterval);
    setTimeout(() => {
    alert("...암전. 당신의 영혼은 이 지옥에 영원히 묶였습니다.");
    location.reload();
    }, 500);
    return true;
    } else if (compMarbles <= 0) {
    clearInterval(timerInterval);
    setTimeout(() => {
    alert("철창문이 열립니다. 당신은 수많은 시체를 등지고 걸어 나갑니다.");
    location.reload();
    }, 500);
    return true;
    }
    return false;
    }
    updateUI();
